cmake_minimum_required(VERSION 3.22.1)
project("mavlink")

# ─── Paths ────────────────────────────────────────────────────────────────────
set(ROUTER_DIR ${CMAKE_CURRENT_SOURCE_DIR}/mavlink-router)
set(ROUTER_SRC ${ROUTER_DIR}/src)
set(ROUTER_COMMON ${ROUTER_SRC}/common)
set(MAVLINK_INC  ${ROUTER_DIR}/modules/mavlink_c_library_v2/ardupilotmega)
set(MAVLINK_ROOT ${ROUTER_DIR}/modules/mavlink_c_library_v2)

# ─── Shadow headers directory (takes priority over system/NDK headers) ─────────
# We stub out headers that reference POSIX API not present in Android NDK.
set(GENERATED_DIR ${CMAKE_CURRENT_BINARY_DIR}/generated)
set(SHADOW_DIR    ${CMAKE_CURRENT_BINARY_DIR}/shadow_includes)
file(MAKE_DIRECTORY ${GENERATED_DIR})
file(MAKE_DIRECTORY ${SHADOW_DIR})

# config.h — normally generated by meson
file(WRITE ${GENERATED_DIR}/config.h
"/* Stub config.h for Android NDK build */
#pragma once
#define HAVE_DECL_AIO_INIT 0
#define HAVE_WADDRESS_OF_PACKED_MEMBER 0
#define PACKAGE_VERSION \"android-ndk\"
/* LINE_MAX is not in Android NDK limits.h */
#ifndef LINE_MAX
#define LINE_MAX 2048
#endif
")

# Shadow mavlink.h — placed in GENERATED_DIR which is first in the include path.
# When src/common/mavlink.h does #include <mavlink.h>, the compiler finds this
# file before ardupilotmega/mavlink.h. We forward to ardupilotmega/ardupilotmega.h
# which is the full dialect header (includes all message types, including
# mavlink_remote_log_data_block_t which binlog.h requires at line 50).
file(WRITE ${GENERATED_DIR}/mavlink.h
"/* Shadow mavlink.h: forces the full ArduPilot MAVLink dialect for mavlink-router.
   When src/common/mavlink.h does #include <mavlink.h>, this file is found first
   via GENERATED_DIR being first in the include path.
   We forward to ardupilotmega/mavlink.h which is the proper dialect bootstrapper.
   MAVLINK_ROOT (parent of ardupilotmega/) must be in the include path for this to work. */
#pragma once
#pragma GCC diagnostic push
#pragma GCC diagnostic ignored \"-Waddress-of-packed-member\"
#include \"ardupilotmega/mavlink.h\"
#pragma GCC diagnostic pop
")


# git_version.h — normally generated by meson vcs_tag
file(WRITE ${GENERATED_DIR}/git_version.h
"#pragma once
static const char *git_version = \"android-ndk-build\";
")

# aio.h stub — Android NDK does not have POSIX AIO.
# mavlink-router only uses it for logendpoint (flight log writing).
# We are not using log endpoints, so stub out the types.
file(WRITE ${SHADOW_DIR}/aio.h
"/* Android NDK stub: POSIX AIO not available */
#pragma once
#ifndef _AIO_H
#define _AIO_H
typedef struct aiocb { int aio_fildes; int aio_lio_opcode; volatile void *aio_buf; int aio_nbytes; struct sigevent aio_sigevent; int __next_prio; int __abs_prio; int __policy; int __error_code; long __return_value; off_t aio_offset; } aiocb;
static inline int aio_write(struct aiocb *) { return -1; }
static inline int aio_return(struct aiocb *) { return -1; }
static inline int aio_error(const struct aiocb *) { return EINVAL; }
static inline int aio_fsync(int, struct aiocb *) { return -1; }
#endif
")


# mavlink_remote_log_data_block_t is an ArduPilot-specific MAVLink message type
# used only in binlog.h. We provide a stub so the header compiles without the
# full ardupilot dialect headers.
file(WRITE ${SHADOW_DIR}/mavlink_stub_ardupilot.h
"#pragma once
typedef struct { unsigned char payload[200]; } mavlink_remote_log_data_block_t;
")

# ─── mavlink-router core static library ───────────────────────────────────────
# We include endpoint.cpp, mainloop.cpp, dedup.cpp, pollable.cpp, timeout.cpp
# and the common utilities. We deliberately EXCLUDE:
#   - main.cpp         (has main() entry point)
#   - binlog.cpp       (binary flight log — needs full AIO, not used on VTX)
#   - logendpoint.cpp  (parent class for log endpoints)
#   - tlog.cpp / ulog.cpp / autolog.cpp  (other log endpoints)
#   - xtermios.cpp     (uses Linux-specific termios struct with c_cc[19], Android has cc[32])
#   - *_test.cpp       (unit tests)
add_library(mavlink_router_core STATIC
    ${ROUTER_SRC}/dedup.cpp
    ${ROUTER_SRC}/endpoint.cpp
    ${ROUTER_SRC}/mainloop.cpp
    ${ROUTER_SRC}/pollable.cpp
    ${ROUTER_SRC}/timeout.cpp
    # Log endpoint classes — included so vtables + symbols link correctly.
    # We don't USE flight logging on VTX, but mainloop.cpp references these types.
    # The aio.h stub makes them compile; disk write calls return -1 (no-op).
    ${ROUTER_SRC}/logendpoint.cpp
    ${ROUTER_SRC}/binlog.cpp
    ${ROUTER_SRC}/tlog.cpp
    ${ROUTER_SRC}/ulog.cpp
    ${ROUTER_SRC}/autolog.cpp
    ${ROUTER_COMMON}/conf_file.cpp
    ${ROUTER_COMMON}/log.cpp
    ${ROUTER_COMMON}/util.cpp
    # xtermios.cpp excluded: Android termios c_cc size differs from Linux assumption
)


target_include_directories(mavlink_router_core PUBLIC
    ${GENERATED_DIR}         # <<< FIRST: config.h (auto-included) and shadow mavlink.h
    ${SHADOW_DIR}            # shadow aio.h
    ${ROUTER_DIR}/src        # endpoint.h, mainloop.h etc.
    ${MAVLINK_ROOT}          # parent of ardupilotmega/ — so ardupilotmega/ardupilotmega.h resolves
    ${MAVLINK_INC}           # ardupilotmega dialect headers for direct includes
)

target_compile_options(mavlink_router_core PRIVATE
    -include ${GENERATED_DIR}/config.h   # auto-include config.h (matches meson)
    -Wno-unused-parameter
    -Wno-unused-result
    -Wno-missing-field-initializers
    -Wno-address-of-packed-member
    -Wno-format-nonliteral
    -Wno-sign-compare
    -DANDROID_BUILD=1
    -DNO_LOG_ENDPOINT=1      # diagnostic: signals we skipped log endpoints
)

target_link_libraries(mavlink_router_core
    log
)

# ─── Main shared library (JNI entry points) ───────────────────────────────────
add_library(${CMAKE_PROJECT_NAME} SHARED
    mavlink.cpp
    mavlink_router_jni.cpp
    xtermios_stub.cpp       # stub for reset_uart() — xtermios.cpp excluded (Android termios size mismatch)
)


target_include_directories(${CMAKE_PROJECT_NAME} PRIVATE
    ${GENERATED_DIR}                        # shadow mavlink.h + config.h (auto-included)
    ${SHADOW_DIR}                           # shadow headers first
    ${CMAKE_CURRENT_SOURCE_DIR}             # mavlink.h (our own)
    ${CMAKE_CURRENT_SOURCE_DIR}/mavlink     # our existing mavlink C headers subdir
    ${ROUTER_DIR}/src                       # mainloop.h, endpoint.h
    ${MAVLINK_ROOT}                         # parent of ardupilotmega/
    ${MAVLINK_INC}                          # ardupilotmega dialect headers
)

target_compile_options(${CMAKE_PROJECT_NAME} PRIVATE
    -include ${GENERATED_DIR}/config.h
    -Wno-address-of-packed-member
    -Wno-missing-field-initializers
)

target_link_libraries(${CMAKE_PROJECT_NAME}
    mavlink_router_core
    android
    log
)
